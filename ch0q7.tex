\begin{question}[Ewing]
  $(A \cup B) \setminus B = A$ is false
\end{question}

\begin{proof}
  Let $A=\{1, 2\}$ and $B=\{2, 3\}$. Then $A\cup B=\{1,2,3\}$ and
  $$(A\cup B)-B=\{1,2,3\}-\{2,3\}=\{1\}\ne A$$
\end{proof}
